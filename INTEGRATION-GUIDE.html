Feature 1: Dynamic Meta Tags

Purpose: Automatically generate and manage Open Graph, Twitter Cards, and meta descriptions for optimal sharing and SEO performance.

const useSEOMetaTags = (article: ArticleData) => { const metaTags = { 'og:title': article.title, 'og:description': article.excerpt, 'og:image': article.featuredImage, 'og:url': article.url, 'twitter:card': 'summary_large_image', 'twitter:title': article.title, 'description': article.metaDescription }; useEffect(() => { Object.entries(metaTags).forEach(([key, value]) => { const meta = document.createElement('meta'); meta.setAttribute(key, value); document.head.appendChild(meta); }); }, [article]); return metaTags; };

Implementation Details:

  • Automatically update meta tags on content changes
  • Support for Open Graph (Facebook), Twitter Cards, and standard meta tags
  • Image optimization for different platforms (1200x630 for OG, 1024x512 for Twitter)
  • Character limit enforcement for descriptions (155-160 chars)
  • Return: Object with all generated meta tag properties
Note: Ensure all images meet platform-specific dimensions for optimal display in social shares.

Feature 2: Canonical URL Management

Purpose: Prevent duplicate content issues and consolidate search authority across content variations.

const useCanonicalURL = (url: string) => { useEffect(() => { let canonicalTag = document.querySelector('link[rel="canonical"]'); if (!canonicalTag) { canonicalTag = document.createElement('link'); canonicalTag.rel = 'canonical'; document.head.appendChild(canonicalTag); } canonicalTag.href = url; }, [url]); };

Implementation Details:

  • Set canonical URL for every page to consolidate authority
  • Handle URL parameters and variations systematically
  • Support both absolute and relative URL formats
  • Remove duplicate canonical tags
Warning: Self-referential canonical URLs must point to the primary version of the content.

Feature 3: Sitemap XML Generation

Purpose: Auto-generate dynamic XML sitemaps for search engine discovery.

const useSitemapXML = (articles: ArticleData[]) => { useEffect(() => { const sitemap = `<?xml version="1.0" encoding="UTF-8"?> <urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9"> ${articles.map(article => ` <url> <loc>${article.url}</loc> <lastmod>${article.updatedAt}</lastmod> <changefreq>weekly</changefreq> <priority>${article.featured ? '0.8' : '0.5'}</priority> </url> `).join('')} </urlset>`; // Serve at /sitemap.xml }, [articles]); };

Implementation Details:

  • Generate XML sitemap dynamically from article database
  • Include lastmod timestamps for freshness signals
  • Set priority values based on article importance
  • Support sitemap index for large sites (>50k URLs)
  • Serve at /sitemap.xml endpoint

Feature 5: Reading Time Estimator

Purpose: Display estimated reading time to improve user engagement.

const useReadTime = (content: string) => { const [readTime, setReadTime] = React.useState(0); useEffect(() => { const wordCount = content.split(/\s+/).length; const readingSpeed = 200; // words per minute const time = Math.ceil(wordCount / readingSpeed); setReadTime(time); }, [content]); return { readTime, wordCount: content.split(/\s+/).length }; };

Implementation Details:

  • Calculate read time based on average reading speed (200 WPM)
  • Display prominently near article title
  • Return both read time and word count
  • Round up for accuracy

Feature 7: Scroll Depth Tracking

Purpose: Track content consumption depth.

const useScrollDepthTracking = () => { useEffect(() => { const handleScroll = () => { const scrolled = (window.scrollY / (document.documentElement.scrollHeight - window.innerHeight)) * 100; if (scrolled > 25 && !tracked.includes(25)) ga('event', 'scroll_depth', { depth: 25 }); if (scrolled > 50 && !tracked.includes(50)) ga('event', 'scroll_depth', { depth: 50 }); if (scrolled > 75 && !tracked.includes(75)) ga('event', 'scroll_depth', { depth: 75 }); if (scrolled > 100) ga('event', 'scroll_depth', { depth: 100 }); }; window.addEventListener('scroll', handleScroll); return () => window.removeEventListener('scroll', handleScroll); }, []); };

Implementation Details:

  • Track scroll depth at 25%, 50%, 75%, 100%
  • Only log each milestone once per page load
  • Send GA4 events with depth data
  • Identify user engagement patterns

Feature 8: Engagement Scoring

const useEngagementScore = (interactions: Interaction[]) => { const calculateScore = () => { let score = 0; score += interactions.filter(i => i.type === 'click').length * 5; score += interactions.filter(i => i.type === 'scroll').length * 2; score += interactions.filter(i => i.type === 'share').length * 50; score += interactions.filter(i => i.type === 'comment').length * 30; return Math.min(score, 100); }; return { score: calculateScore(), level: score > 75 ? Server-Side Changes

Express.js configuration and endpoints required to support the integrated features.

POST /api/analytics/events

Ingest analytics events from the client and forward to GA4.

app.post('/api/analytics/events', async (req, res) => { const { eventName, eventData, userId } = req.body; // Forward to GA4 Measurement Protocol await fetch('https://www.google-analytics.com/mp/collect', { method: 'POST', body: JSON.stringify({ client_id: userId, events: [{ name: eventName, params: eventData }] }) }); res.json({ success: true }); });

GET /sitemap.xml

Generate and serve dynamic XML sitemap.

app.get('/sitemap.xml', async (req, res) => { const articles = await db.getAllArticles(); const sitemap = `<?xml version="1.0" encoding="UTF-8"?> <urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9"> ${articles.map(a => `<url><loc>${a.url}</loc></url>`).join('')} </urlset>`; res.type('application/xml').send(sitemap); });

GET /robots.txt

Serve robots.txt for search engine crawlers.

app.get('/robots.txt', (req, res) => { const robots = `User-agent: * Allow: / Disallow: /admin/ Disallow: /search?* Sitemap: ${process.env.SITE_URL}/sitemap.xml`; res.type('text/plain').send(robots); });

POST /api/newsletter/subscribe

Handle email subscription and integration with email service.

app.post('/api/newsletter/subscribe', async (req, res) => { const { email } = req.body; // Example: Mailchimp integration await fetch(`https://us1.api.mailchimp.com/3.0/lists/${MAILCHIMP_LIST_ID}/members`, { method: 'POST', headers: { 'Authorization': `Basic ${Buffer.from(`anything:${MAILCHIMP_API_KEY}`).toString('base64')}` }, body: JSON.stringify({ email_address: email, status: 'pending' }) }); res.json({ success: true }); });
